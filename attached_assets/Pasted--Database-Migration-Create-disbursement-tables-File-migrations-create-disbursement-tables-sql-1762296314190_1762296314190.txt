-- Create disbursements table
CREATE TABLE IF NOT EXISTS disbursements (
  id SERIAL PRIMARY KEY,
  law_firm_id INTEGER NOT NULL REFERENCES law_firms(id) ON DELETE CASCADE,
  client_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  client_amount DECIMAL(10, 2) NOT NULL,
  medical_total DECIMAL(10, 2) DEFAULT 0,
  platform_fee DECIMAL(10, 2) NOT NULL DEFAULT 200.00,
  total_amount DECIMAL(10, 2) NOT NULL,
  stripe_payment_intent_id VARCHAR(255),
  stripe_transfer_id VARCHAR(255),
  status VARCHAR(50) DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(client_id, status) WHERE status = 'completed' -- Only one completed disbursement per client
);

-- Create disbursement_medical_payments table
CREATE TABLE IF NOT EXISTS disbursement_medical_payments (
  id SERIAL PRIMARY KEY,
  disbursement_id INTEGER NOT NULL REFERENCES disbursements(id) ON DELETE CASCADE,

  amount DECIMAL(10, 2) NOT NULL,

  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP

-- Create view for easy disbursement reporting
CREATE OR REPLACE VIEW disbursement_summary AS
SELECT 
  d.id,
  d.law_firm_id,
  lf.firm_name,
  d.client_id,
  CONCAT(u.first_name, ' ', u.last_name) as client_name,
  u.email as client_email,
  d.client_amount,
  d.medical_total,
  d.platform_fee,
  d.total_amount,
  d.status,
  d.created_at,
  COUNT(dmp.id) as medical_provider_count
FROM disbursements d
JOIN law_firms lf ON d.law_firm_id = lf.id
JOIN users u ON d.client_id = u.id
LEFT JOIN disbursement_medical_payments dmp ON d.id = dmp.disbursement_id
GROUP BY d.id, lf.firm_name, u.first_name, u.last_name, u.email;

COMMENT ON COLUMN disbursements.stripe_payment_intent_id IS 'Stripe Payment Intent ID for charging the law firm';
COMMENT ON COLUMN disbursements.stripe_transfer_id IS 'Stripe Transfer ID for paying the client';
