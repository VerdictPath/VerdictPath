import React, { useState } from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet, ScrollView, Alert } from 'react-native';
import { useNavigation } from '@react-navigation/native';
import AsyncStorage from '@react-native-async-storage/async-storage';

const API_URL = 'https://your-railway-url.up.railway.app'; // UPDATE THIS

export default function LawFirmRegistrationScreen() {
    const navigation = useNavigation();
    const [mode, setMode] = useState(null); // 'create' or 'join'
    
    // Create new firm fields
    const [firmName, setFirmName] = useState('');
    const [adminFirstName, setAdminFirstName] = useState('');
    const [adminLastName, setAdminLastName] = useState('');
    
    // Join existing firm fields
    const [firmCode, setFirmCode] = useState('');
    const [firstName, setFirstName] = useState('');
    const [lastName, setLastName] = useState('');
    const [requestedRole, setRequestedRole] = useState('attorney');
    
    const [loading, setLoading] = useState(false);

    const handleCreateNewFirm = async () => {
        if (!firmName.trim() || !adminFirstName.trim() || !adminLastName.trim()) {
            Alert.alert('Missing Information', 'Please fill in all fields');
            return;
        }

        setLoading(true);
        try {
            const token = await AsyncStorage.getItem('token');
            
            const response = await fetch(`${API_URL}/api/lawfirm/register-new-firm`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': `Bearer ${token}`
                },
                body: JSON.stringify({
                    firm_name: firmName,
                    admin_first_name: adminFirstName,
                    admin_last_name: adminLastName
                })
            });

            const data = await response.json();

            if (response.ok) {
                Alert.alert(
                    'üè¥‚Äç‚ò†Ô∏è Firm Created!',
                    `${data.firm.firm_name}\n\nYour Firm Code: ${data.firm.firm_code}\n\nShare this code with your team members so they can join your firm!\n\nüéâ SPECIAL OFFER: Only $40/month for unlimited users!`,
                    [
                        {
                            text: 'Continue to Dashboard',
                            onPress: () => navigation.replace('LawFirmDashboard')
                        }
                    ]
                );
            } else {
                Alert.alert('Error', data.error || 'Failed to create firm');
            }
        } catch (error) {
            console.error('Error creating firm:', error);
            Alert.alert('Error', 'Failed to create firm. Please try again.');
        } finally {
            setLoading(false);
        }
    };

    const handleJoinExistingFirm = async () => {
        if (!firmCode.trim() || !firstName.trim() || !lastName.trim()) {
            Alert.alert('Missing Information', 'Please fill in all fields');
            return;
        }

        setLoading(true);
        try {
            const token = await AsyncStorage.getItem('token');
            
            const response = await fetch(`${API_URL}/api/lawfirm/join-existing-firm`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': `Bearer ${token}`
                },
                body: JSON.stringify({
                    firm_code: firmCode.toUpperCase(),
                    first_name: firstName,
                    last_name: lastName,
                    requested_role: requestedRole
                })
            });

            const data = await response.json();

            if (response.ok) {
                Alert.alert(
                    '‚è≥ Request Sent!',
                    `Your request to join ${data.firm_name} has been sent to the firm administrators.\n\nYou'll be notified when an admin approves your request.`,
                    [
                        {
                            text: 'OK',
                            onPress: () => navigation.replace('PendingApprovalScreen')
                        }
                    ]
                );
            } else {
                Alert.alert('Error', data.error || 'Failed to join firm');
            }
        } catch (error) {
            console.error('Error joining firm:', error);
            Alert.alert('Error', 'Failed to join firm. Please try again.');
        } finally {
            setLoading(false);
        }
    };

    const renderPricingCard = () => (
        <View style={styles.pricingSection}>
            <View style={styles.pricingHeader}>
                <Text style={styles.pricingBadge}>üéâ LIMITED TIME OFFER</Text>
            </View>
            
            <Text style={styles.pricingTitle}>Law Firm Portal Pricing</Text>
            
            {/* Tier Breakdown with Crossed Out Prices */}
            <View style={styles.tierContainer}>
                <View style={styles.tierCard}>
                    <Text style={styles.tierName}>Startup Firms</Text>
                    <Text style={styles.tierRange}>1-5 Clients</Text>
                    <View style={styles.priceRow}>
                        <Text style={styles.originalPrice}>$45/mo</Text>
                        <Text style={styles.salePrice}>$40/mo</Text>
                    </View>
                </View>

                <View style={styles.tierCard}>
                    <Text style={styles.tierName}>Small Firms</Text>
                    <Text style={styles.tierRange}>6-20 Clients</Text>
                    <View style={styles.priceRow}>
                        <Text style={styles.originalPrice}>$99/mo</Text>
                        <Text style={styles.salePrice}>$40/mo</Text>
                    </View>
                </View>

                <View style={styles.tierCard}>
                    <Text style={styles.tierName}>Medium Firms</Text>
                    <Text style={styles.tierRange}>21-50 Clients</Text>
                    <View style={styles.priceRow}>
                        <Text style={styles.originalPrice}>$199/mo</Text>
                        <Text style={styles.salePrice}>$40/mo</Text>
                    </View>
                </View>

                <View style={styles.tierCard}>
                    <Text style={styles.tierName}>Large Firms</Text>
                    <Text style={styles.tierRange}>51-100 Clients</Text>
                    <View style={styles.priceRow}>
                        <Text style={styles.originalPrice}>$399/mo</Text>
                        <Text style={styles.salePrice}>$40/mo</Text>
                    </View>
                </View>

                <View style={styles.tierCard}>
                    <Text style={styles.tierName}>Enterprise Firms</Text>
                    <Text style={styles.tierRange}>101-200 Clients</Text>
                    <View style={styles.priceRow}>
                        <Text style={styles.originalPrice}>$799/mo</Text>
                        <Text style={styles.salePrice}>$40/mo</Text>
                    </View>
                </View>

                <View style={styles.tierCard}>
                    <Text style={styles.tierName}>Premier Firms</Text>
                    <Text style={styles.tierRange}>201-500 Clients</Text>
                    <View style={styles.priceRow}>
                        <Text style={styles.originalPrice}>$1,299/mo</Text>
                        <Text style={styles.salePrice}>$40/mo</Text>
                    </View>
                </View>

                <View style={styles.tierCard}>
                    <Text style={styles.tierName}>Elite Firms</Text>
                    <Text style={styles.tierRange}>501-1000 Clients</Text>
                    <View style={styles.priceRow}>
                        <Text style={styles.originalPrice}>$1,599/mo</Text>
                        <Text style={styles.salePrice}>$40/mo</Text>
                    </View>
                </View>

                <View style={[styles.tierCard, styles.tierCardHighlight]}>
                    <Text style={styles.tierName}>Mega Firms</Text>
                    <Text style={styles.tierRange}>1001+ Clients</Text>
                    <View style={styles.priceRow}>
                        <Text style={styles.originalPrice}>$1,800/mo</Text>
                        <Text style={styles.salePrice}>$40/mo</Text>
                    </View>
                </View>
            </View>

            <View style={styles.savingsBox}>
                <Text style={styles.savingsText}>üí∞ SAVE UP TO $1,760/MONTH!</Text>
                <Text style={styles.savingsSubtext}>One flat rate ‚Ä¢ Unlimited users ‚Ä¢ All features included</Text>
            </View>

            <View style={styles.featuresList}>
                <Text style={styles.featuresTitle}>‚ú® What's Included:</Text>
                <Text style={styles.featureItem}>‚úì Unlimited team members</Text>
                <Text style={styles.featureItem}>‚úì Client management dashboard</Text>
                <Text style={styles.featureItem}>‚úì Real-time case tracking</Text>
                <Text style={styles.featureItem}>‚úì Secure document access</Text>
                <Text style={styles.featureItem}>‚úì Admin user controls</Text>
                <Text style={styles.featureItem}>‚úì Activity logging & reporting</Text>
            </View>

            <View style={styles.urgencyBox}>
                <Text style={styles.urgencyText}>‚è∞ Launch Special - Don't Miss Out!</Text>
                <Text style={styles.urgencySubtext}>Lock in this rate before it goes up</Text>
            </View>
        </View>
    );

    if (!mode) {
        return (
            <View style={styles.container}>
                <ScrollView contentContainerStyle={styles.scrollContent}>
                    <Text style={styles.title}>‚öì Law Firm Setup</Text>

                        style={styles.optionCard}

    if (mode === 'create') {
        return (

                    <TouchableOpacity 
                        style={styles.backButton}
                        onPress={() => setMode(null)}

                        <Text style={styles.backButtonText}>‚Üê Back</Text>

                    <Text style={styles.subtitle}>
                        You'll become the first administrator
                    </Text>

                    <View style={styles.priceCallout}>
                        <Text style={styles.priceCalloutText}>
                            üéâ Only $40/month for your entire firm!
                        </Text>
                    </View>

                    <View style={styles.form}>
                        <Text style={styles.label}>Law Firm Name</Text>
                        <TextInput
                            style={styles.input}
                            placeholder="e.g., Smith & Associates"
                            value={firmName}
                            onChangeText={setFirmName}
                            autoCapitalize="words"
                        />

                        <Text style={styles.label}>Your First Name</Text>
                        <TextInput
                            style={styles.input}
                            placeholder="First Name"
                            value={adminFirstName}
                            onChangeText={setAdminFirstName}
                            autoCapitalize="words"
                        />

                        <Text style={styles.label}>Your Last Name</Text>
                        <TextInput
                            style={styles.input}
                            placeholder="Last Name"
                            value={adminLastName}
                            onChangeText={setAdminLastName}
                            autoCapitalize="words"
                        />

                        <TouchableOpacity
                            style={[styles.button, loading && styles.buttonDisabled]}
                            onPress={handleCreateNewFirm}
                            disabled={loading}
                        >
                            <Text style={styles.buttonText}>
                                {loading ? 'Creating Firm...' : 'Create Firm & Get Code'}
                            </Text>
                        </TouchableOpacity>
                    </View>
                </ScrollView>
            </View>
        );
    }

    // mode === 'join'
    return (
        <View style={styles.container}>
            <ScrollView contentContainerStyle={styles.scrollContent}>
                <TouchableOpacity 
                    style={styles.backButton}
                    onPress={() => setMode(null)}
                >
                    <Text style={styles.backButtonText}>‚Üê Back</Text>
                </TouchableOpacity>

                <Text style={styles.title}>‚õµ Join Existing Firm</Text>
                <Text style={styles.subtitle}>
                    Enter the firm code provided by your administrator
                </Text>

                <View style={styles.form}>
                    <Text style={styles.label}>Firm Code</Text>
                    <TextInput
                        style={[styles.input, styles.codeInput]}
                        placeholder="VPL-XXXX"
                        value={firmCode}
                        onChangeText={(text) => setFirmCode(text.toUpperCase())}
                        autoCapitalize="characters"
                        maxLength={8}
                    />

                    <Text style={styles.label}>Your First Name</Text>
                    <TextInput
                        style={styles.input}
                        placeholder="First Name"
                        value={firstName}
                        onChangeText={setFirstName}
                        autoCapitalize="words"
                    />

                    <Text style={styles.label}>Your Last Name</Text>
                    <TextInput
                        style={styles.input}
                        placeholder="Last Name"
                        value={lastName}
                        onChangeText={setLastName}
                        autoCapitalize="words"
                    />

                    <Text style={styles.label}>Your Role</Text>
                    <View style={styles.roleButtons}>
                        {['attorney', 'paralegal', 'staff'].map((role) => (
                            <TouchableOpacity
                                key={role}
                                style={[
                                    styles.roleButton,
                                    requestedRole === role && styles.roleButtonActive
                                ]}
                                onPress={() => setRequestedRole(role)}
                            >
                                <Text style={[
                                    styles.roleButtonText,
                                    requestedRole === role && styles.roleButtonTextActive
                                ]}>
                                    {role.charAt(0).toUpperCase() + role.slice(1)}
                                </Text>
                            </TouchableOpacity>
                        ))}
                    </View>

                    <TouchableOpacity
                        style={[styles.button, loading && styles.buttonDisabled]}
                        onPress={handleJoinExistingFirm}
                        disabled={loading}
                    >
                        <Text style={styles.buttonText}>
                            {loading ? 'Sending Request...' : 'Request to Join Firm'}
                        </Text>
                    </TouchableOpacity>

                    <Text style={styles.note}>
                        üí° A firm administrator will review and approve your request
                    </Text>
                </View>
            </ScrollView>
        </View>
    );
}
