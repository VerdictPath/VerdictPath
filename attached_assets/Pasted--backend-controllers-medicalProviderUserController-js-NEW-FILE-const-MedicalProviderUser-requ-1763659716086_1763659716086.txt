const MedicalProviderUser = require('../models/MedicalProviderUser');

const MedicalActivityLog = require('../models/MedicalActivityLog');
const jwt = require('jsonwebtoken');
const { createMedicalActivityLog } = require('../middleware/medicalActivityLogger');

// Generate unique user code
const generateUserCode = async (providerCode) => {
  let userCode;
  let exists = true;
  
  while (exists) {
    const randomNum = Math.floor(1000 + Math.random() * 9000);
    userCode = `${providerCode}-USER-${randomNum}`;
    
    const existing = await MedicalProviderUser.findOne({ userCode });
    exists = !!existing;
  }
  
  return userCode;
};

// Admin creates new user
exports.createUser = async (req, res) => {
  try {
    const {
      firstName,
      lastName,
      email,
      password,
      role,
      permissions,
      credentials,
      licenseNumber,
      npiNumber,
      title,
      phoneNumber,
      department,
      specialties,
      hipaaTrainingDate,
    } = req.body;

    // Get medical provider
    const provider = await MedicalProvider.findById(req.medicalProviderId);
    
    if (!provider) {
      return res.status(404).json({
        success: false,
        message: 'Medical provider not found',
      });
    }

    // Check user limit based on subscription
    const currentUserCount = await MedicalProviderUser.countDocuments({
      medicalProviderId: req.medicalProviderId,
      status: 'active',
    });

    if (currentUserCount >= provider.settings.maxUsers) {
      return res.status(400).json({
        success: false,
        message: `User limit reached. Your ${provider.subscriptionTier} plan allows ${provider.settings.maxUsers} users.`,
        currentUsers: currentUserCount,
        maxUsers: provider.settings.maxUsers,
      });
    }

    // Check if email already exists
    const existingUser = await MedicalProviderUser.findOne({ email: email.toLowerCase() });
    if (existingUser) {
      return res.status(400).json({
        success: false,
        message: 'Email already registered',
      });
    }

    // Generate unique user code
    const userCode = await generateUserCode(provider.providerCode);

    // Set default permissions based on role
    let finalPermissions = permissions || {};
    if (role === 'admin') {
      finalPermissions = {
        canManageUsers: true,
        canManagePatients: true,
        canViewAllPatients: true,
        canSendNotifications: true,
        canViewMedicalRecords: true,
        canEditMedicalRecords: true,
        canManageDisbursements: true,
        canViewAnalytics: true,
        canManageSettings: true,
        canAccessPHI: true,
      };
    } else if (role === 'doctor') {
      finalPermissions = {
        canManageUsers: false,
        canManagePatients: true,
        canViewAllPatients: true,
        canSendNotifications: true,
        canViewMedicalRecords: true,
        canEditMedicalRecords: true,
        canManageDisbursements: false,
        canViewAnalytics: true,
        canManageSettings: false,
        canAccessPHI: true,
      };
    }

    // Calculate HIPAA training expiry (typically 1 year)
    let hipaaTrainingExpiry = null;
    if (hipaaTrainingDate) {
      hipaaTrainingExpiry = new Date(hipaaTrainingDate);
      hipaaTrainingExpiry.setFullYear(hipaaTrainingExpiry.getFullYear() + 1);
    }

    // Create user
    const user = await MedicalProviderUser.create({
      medicalProviderId: req.medicalProviderId,
      firstName,
      lastName,
      email: email.toLowerCase(),
      password,
      userCode,
      role: role || 'doctor',
      permissions: finalPermissions,
      credentials,
      licenseNumber,
      npiNumber,
      title,
      phoneNumber,
      department,
      specialties: specialties || [],
      hipaaTrainingDate: hipaaTrainingDate || new Date(),
      hipaaTrainingExpiry,
      createdBy: req.user._id,
    });

    // Log activity
    await createMedicalActivityLog({
      medicalProviderId: req.medicalProviderId,
      userId: req.user._id,
      userEmail: req.user.email,
      userName: `${req.user.firstName} ${req.user.lastName}`,
      userRole: req.user.role,
      action: 'user_created',
      actionCategory: 'user',
      sensitivityLevel: 'medium',
      targetType: 'MedicalProviderUser',
      targetId: user._id,
      targetName: `${user.firstName} ${user.lastName}`,
      metadata: {
        role: user.role,
        userCode: user.userCode,
        credentials: user.credentials,
      },
      ipAddress: req.ip,
      status: 'success',
      hipaaCompliant: true,
    });

    res.status(201).json({
      success: true,
      message: 'User created successfully',
      user: {
        id: user._id,
        firstName: user.firstName,
        lastName: user.lastName,
        email: user.email,
        userCode: user.userCode,
        role: user.role,
        credentials: user.credentials,
        permissions: user.permissions,
        status: user.status,
        hipaaTrainingExpiry: user.hipaaTrainingExpiry,
      },
    });

  } catch (error) {
    console.error('[MedicalUserController] Create user error:', error);
    res.status(500).json({
      success: false,
      message: 'Failed to create user',
      error: error.message,
    });
  }
};

// Get all users in provider practice
exports.getAllUsers = async (req, res) => {
  try {
    const { status, role, search } = req.query;

    let query = { medicalProviderId: req.medicalProviderId };

    if (status) {
      query.status = status;
    }

    if (role) {
      query.role = role;
    }

    if (search) {
      query.$or = [
        { firstName: { $regex: search, $options: 'i' } },
        { lastName: { $regex: search, $options: 'i' } },
        { email: { $regex: search, $options: 'i' } },
        { userCode: { $regex: search, $options: 'i' } },
        { credentials: { $regex: search, $options: 'i' } },
      ];
    }

    const users = await MedicalProviderUser.find(query)
      .select('-password')
      .sort({ createdAt: -1 });

    res.json({
      success: true,
      count: users.length,
      users: users,
    });

  } catch (error) {
    console.error('[MedicalUserController] Get users error:', error);
    res.status(500).json({
      success: false,
      message: 'Failed to fetch users',
    });
  }
};

// Get single user details
exports.getUserDetails = async (req, res) => {
  try {
    const { userId } = req.params;

    const user = await MedicalProviderUser.findOne({
      _id: userId,
      medicalProviderId: req.medicalProviderId,
    })
      .select('-password')
      .populate('createdBy', 'firstName lastName email')
      .populate('deactivatedBy', 'firstName lastName email');

    if (!user) {
      return res.status(404).json({
        success: false,
        message: 'User not found',
      });
    }

    // Get user's recent activity
    const recentActivity = await MedicalActivityLog.find({
      userId: user._id,
    })
      .sort({ timestamp: -1 })
      .limit(20);

    res.json({
      success: true,
      user: user,
      recentActivity: recentActivity,
    });

  } catch (error) {
    console.error('[MedicalUserController] Get user details error:', error);
    res.status(500).json({
      success: false,
      message: 'Failed to fetch user details',
    });
  }
};

// Update user
exports.updateUser = async (req, res) => {
  try {
    const { userId } = req.params;
    const updates = req.body;

    // Don't allow changing certain fields
    delete updates.medicalProviderId;
    delete updates.userCode;
    delete updates.createdBy;
    delete updates.createdAt;

    const user = await MedicalProviderUser.findOneAndUpdate(
      { _id: userId, medicalProviderId: req.medicalProviderId },
      { ...updates, updatedAt: Date.now() },
      { new: true, runValidators: true }
    ).select('-password');

    if (!user) {
      return res.status(404).json({
        success: false,
        message: 'User not found',
      });
    }

    // Log activity
    await createMedicalActivityLog({
      medicalProviderId: req.medicalProviderId,
      userId: req.user._id,
      userEmail: req.user.email,
      userName: `${req.user.firstName} ${req.user.lastName}`,
      userRole: req.user.role,
      action: 'user_updated',
      actionCategory: 'user',
      sensitivityLevel: 'medium',
      targetType: 'MedicalProviderUser',
      targetId: user._id,
      targetName: `${user.firstName} ${user.lastName}`,
      metadata: { updates },
      ipAddress: req.ip,
      status: 'success',
      hipaaCompliant: true,
    });

    res.json({
      success: true,
      message: 'User updated successfully',
      user: user,
    });

  } catch (error) {
    console.error('[MedicalUserController] Update user error:', error);
    res.status(500).json({
      success: false,
      message: 'Failed to update user',
    });
  }
};

// Deactivate user (soft delete)
exports.deactivateUser = async (req, res) => {
  try {
    const { userId } = req.params;
    const { reason } = req.body;

    // Can't deactivate yourself
    if (userId === req.user._id.toString()) {
      return res.status(400).json({
        success: false,
        message: 'Cannot deactivate your own account',
      });
    }

    const user = await MedicalProviderUser.findOneAndUpdate(
      { _id: userId, medicalProviderId: req.medicalProviderId },
      {
        status: 'deactivated',
        deactivatedAt: Date.now(),
        deactivatedBy: req.user._id,
        deactivationReason: reason,
      },
      { new: true }
    ).select('-password');

    if (!user) {
      return res.status(404).json({
        success: false,
        message: 'User not found',
      });
    }

    // Log activity (HIGH sensitivity - removing access to PHI)
    await createMedicalActivityLog({
      medicalProviderId: req.medicalProviderId,
      userId: req.user._id,
      userEmail: req.user.email,
      userName: `${req.user.firstName} ${req.user.lastName}`,
      userRole: req.user.role,
      action: 'user_deactivated',
      actionCategory: 'user',
      sensitivityLevel: 'high',
      targetType: 'MedicalProviderUser',
      targetId: user._id,
      targetName: `${user.firstName} ${user.lastName}`,
      metadata: { reason },
      ipAddress: req.ip,
      status: 'success',
      hipaaCompliant: true,
      auditRequired: true,
    });

    res.json({
      success: true,
      message: 'User deactivated successfully',
      user: user,
    });

  } catch (error) {
    console.error('[MedicalUserController] Deactivate user error:', error);
    res.status(500).json({
      success: false,
      message: 'Failed to deactivate user',
    });
  }
};
